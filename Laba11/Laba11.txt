    <button id="btnNoop">$.noop()</button>

<script>
$(document).ready(function(){

    // Завдання 1
    $('#btnHtml').click(function(){
        $('#div1').html('Новий текст у div');
    });

    $('#btnCss').click(function(){
        $('#div1').css({'color':'red','font-weight':'bold'});
    });

    $('#btnVal').click(function(){
        alert('Значення input: ' + $('#input1').val());
    });

    $('#btnAttrGet').click(function(){
        alert('type input: ' + $('#input1').attr('type'));
    });

    $('#btnAttrSet').click(function(){
        $('#input1').attr('placeholder','Введіть текст тут');
    });

    $('#btnAttrObj').click(function(){
        $('#input1').attr({'title':'Тут підказка', 'maxlength':'20'});
    });

    $('#btnAttrFunc').click(function(){
        $('#input1').attr('value', function(i, val){
            return val + ' + додано';
        });
    });

    $('#btnRemoveAttr').click(function(){
        $('#input1').removeAttr('maxlength');
        alert('Атрибут maxlength видалено');
    });

    // Завдання 2
    $('#btnBefore').click(function(){
        $('#div2').before('<p>Текст перед div</p>');
    });

    $('#btnReplaceWith').click(function(){
        $('#div2').replaceWith('<div style="background: pink;">Замінений div</div>');
    });

    $('#btnWidth').click(function(){
        $('div').width(function(i, oldWidth){
            return oldWidth + 20;
        });
    });

    $('#btnPosition').click(function(){
        var pos = $('#div2').position();
        alert('Top: ' + pos.top + ', Left: ' + pos.left);
    });

    // Завдання 3
    $('#btnContents').click(function(){
        console.log($('#list3').contents());
        alert('Подивіться консоль для результату .contents()');
    });

    $('#btnMap').click(function(){
        var texts = $('#list3 li').map(function(){ return $(this).text(); }).get();
        alert('Результат .map(): ' + texts.join(', '));
    });

    $('#btnPrev').click(function(){
        alert('Попередній елемент: ' + $('#list3 li:last').prev().text());
    });

    $('#btnEach').click(function(){
        $('#list3 li').each(function(i){
            $(this).css('color', i%2==0 ? 'green':'blue');
        });
    });

    $('#btnNextAll').click(function(){
        $('#list3 li:first').nextAll().css('background','lightgray');
    });

    // Завдання 4
    $('#btnFilterClass').click(function(){
        $('.item').filter('.item').addClass('modify').text(function(i, txt){ return txt + ' modify'; });
    });

    $('#btnFilterThird').click(function(){
        $('.item').filter(':nth-child(3n)').addClass('modify').text(function(i, txt){ return txt + ' modify'; });
    });

    // Завдання 5
    $('#btnNoop').click($.noop);
});
</script>
