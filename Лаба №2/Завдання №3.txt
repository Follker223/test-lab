<script>
let rows, cols;

function buildMatrix() {
  rows = parseInt(document.getElementById("rows").value);
  cols = parseInt(document.getElementById("cols").value);

  let res_str = "<table border='1' cellspacing='0' cellpadding='5'>\n";
  for (let i = 1; i <= rows; i++) {
    res_str += "<tr>\n";
    for (let j = 1; j <= cols; j++) {
      let value = 7 * j + 2 * i;
      res_str += "<td><input type='text' id='cell_" + i + "_" + j + "' value='" + value + "'></td>\n";
    }
    res_str += "</tr>\n";
  }
  res_str += "</table>";
  document.getElementById("matrix").innerHTML = res_str;
}

function calculate() {
  let sum = 0;
  for (let i = 1; i <= rows; i++) {
    for (let j = 1; j <= cols; j++) {
      sum += parseInt(document.getElementById("cell_" + i + "_" + j).value);
    }
  }
  alert("Сума всіх елементів = " + sum);
}

  <input type="button" value="Побудувати матрицю" onclick="buildMatrix()">
  <input type="button" value="Розрахувати" onclick="calculate()">
