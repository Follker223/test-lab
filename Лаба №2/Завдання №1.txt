  <title>Годинник і календар</title>
</head>
<body onload="clock()">
  <form name="clock">
    <label>Час і дата:</label><br>
    <input type="text" name="datetime" size="25" value="">
  </form>

  <script>
    function clock() {
      let now = new Date();

      let hours = now.getHours();
      let minutes = now.getMinutes();
      let seconds = now.getSeconds();

      hours = (hours < 10 ? "0" : "") + hours;
      minutes = (minutes < 10 ? "0" : "") + minutes;
      seconds = (seconds < 10 ? "0" : "") + seconds;

      let year = now.getFullYear();
      let month = now.getMonth() + 1; // місяці з 0
      let day = now.getDate();

      month = (month < 10 ? "0" : "") + month;
      day = (day < 10 ? "0" : "") + day;

      let dateTimeStr = hours + "$" + minutes + "$" + seconds + "-" +
                        year + "$" + month + "$" + day;

      document.clock.datetime.value = dateTimeStr;

      setTimeout(clock, 1000);
    }
  </script>
