<script>
function checkTest(){
  let form = document.forms['testForm'];
  let total = 0;
  let table = "<table><tr><th>Запитання</th><th>Відповідь</th><th>Бал</th></tr>";

  // === Q1 Radio ===
  let q1 = form.q1.value || "";
  let bal1 = parseInt(q1)||0;
  total += bal1;
  table += `<tr><td>1</td><td>${q1? (bal1?"Київ":"Неправильно"):"не обрано"}</td><td>${bal1}</td></tr>`;

  // === Q2 Radio ===
  let q2 = form.q2.value || "";
  let bal2 = parseInt(q2)||0;
  total += bal2;
  table += `<tr><td>2</td><td>${q2? (bal2?"JavaScript":"Неправильно"):"не обрано"}</td><td>${bal2}</td></tr>`;

  // === Q3 Checkbox ===
  let q3 = [...form.q3].filter(el=>el.checked).map(el=>parseInt(el.value));
  let bal3=0;
  if(q3.length==2){ bal3 = q3[0]+q3[1]; }
  else if(q3.length<2){ bal3 = q3.reduce((a,b)=>a+b,0); }
  total+=bal3;
  table += `<tr><td>3</td><td>${q3.length} обрано</td><td>${bal3}</td></tr>`;

  // === Q4 Checkbox ===
  let q4 = [...form.q4].filter(el=>el.checked).map(el=>parseInt(el.value));
  let bal4=0;
  if(q4.length==2){ bal4 = q4[0]+q4[1]; }
  else if(q4.length<2){ bal4 = q4.reduce((a,b)=>a+b,0); }
  total+=bal4;
  table += `<tr><td>4</td><td>${q4.length} обрано</td><td>${bal4}</td></tr>`;

  // === Q5 Select ===
  let q5 = form.q5.value;
  let bal5 = parseInt(q5);
  total+=bal5;
  table += `<tr><td>5</td><td>${bal5? "Тихий":"Неправильно"}</td><td>${bal5}</td></tr>`;

  // === Q6 Select multiple ===
  let q6 = [...form.q6.options].filter(opt=>opt.selected).map(opt=>parseInt(opt.value));
  let bal6=0;
  if(q6.length==2){ bal6 = q6[0]+q6[1]; }
  else if(q6.length<2){ bal6 = q6.reduce((a,b)=>a+b,0); }
  total+=bal6;
  table += `<tr><td>6</td><td>${q6.length} обрано</td><td>${bal6}</td></tr>`;

  // === Q7 Text ===
  let q7 = form.q7.value.trim();
  let bal7 = (q7==="3.14")?1:0;
  total+=bal7;
  table += `<tr><td>7</td><td>${q7||"не введено"}</td><td>${bal7}</td></tr>`;

  // === Вивід результату ===
  table += `</table><h3>Загальний бал: ${total}</h3>`;
  document.getElementById('result').innerHTML = table;
}
